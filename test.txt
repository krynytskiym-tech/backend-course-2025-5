# Тестування методу GET (картинка відсутня)
curl -i http://127.0.0.1:3000/200

# Тестування методу PUT (завантаження картинки)
curl -i -X PUT http://127.0.0.1:3000/200 --data-binary @test.jpg

# Тестування методу GET (картинка є в кеші)
curl -i http://127.0.0.1:3000/200 -o downloaded.jpg

# Тестування методу DELETE
curl -i -X DELETE http://127.0.0.1:3000/200

# Тестування неправильного методу
curl -i -X POST http://127.0.0.1:3000/200
